#include<iostream>
using namespace std;

struct Node
{
    int data;
    Node *next;
};

Node *head=NULL;

void create()
{
    Node *newnode=new Node;
    cin>>newnode->data;
    newnode->next=NULL;

    if(head==NULL)
        head=newnode;
    else
    {
        Node *temp=head;

        while(temp->next!=NULL)
            temp=temp->next;

        temp->next=newnode;
    }
}

void middle()
{
    Node *slow=head;
    Node *fast=head;

    while(fast!=NULL && fast->next!=NULL)
    {
        slow=slow->next;
        fast=fast->next->next;
    }

    cout<<"Middle Element = "<<slow->data;
}

int main()
{
    int n;

    cin>>n;

    for(int i=0;i<n;i++)
        create();

    middle();

    return 0;
}
