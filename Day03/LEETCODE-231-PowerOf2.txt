#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "Enter N : " << endl;
    cin >> n;
    
    if (n <= 0) {
        cout << "false" << endl;
        return 0; 
    }
    
    while (n % 2 == 0) {
        n = n/2;
    }

    if (n == 1) {
        cout << "true" << endl;
    } else {
        cout << "false" << endl;
    }

    return 0; 
}
